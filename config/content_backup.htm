from browser import html
from browser import document as doc

# 利用 HTML 建立畫布超文件物件
canvas = html.CANVAS(width=400, height=400)
brython_div = doc["brython_div1"]
brython_div <= canvas

# 每格的像素數
gs = 20  # 每個格子的大小

# 取得畫布的 2D 繪圖上下文
ctx = canvas.getContext("2d")

# 繪製矩形的函數
def dRect(lux, luy, w, h, s=1, c='#ff0000'):
    ctx.lineWidth = s
    ctx.strokeStyle = c
    ctx.beginPath()
    ctx.rect(lux, luy, w, h)
    ctx.stroke()

# 繪製 10x10 的格子
def grid(width, height, grid_pix):
    for i in range(width):
        for j in range(height):
            dRect(i * grid_pix, j * grid_pix, grid_pix, grid_pix, 1, "lightgrey")

# 填充格子的函數
def fill(x, y, color):
    ctx.fillStyle = color
    ctx.fillRect(x - 1, y - 1, gs, gs)

# 在方格內繪製圓形
def draw_circle(center_x, center_y, radius, color):
    # 計算圓心的座標在畫布上的位置 (需要考慮格子的大小)
    cx = center_x * gs + gs / 2
    cy = center_y * gs + gs / 2
    
    ctx.fillStyle = color
    ctx.beginPath()
    ctx.arc(cx, cy, radius * gs, 0, 2 * 3.14159)  # 設定圓心座標與半徑
    ctx.fill()

# 繪製 10x10 的格子
grid(10, 10, gs)

# 繪製圓形，圓心在 (5,5)，半徑為 3
draw_circle(5, 5, 3, "black")
